```python
import sys
from collections import deque

sys.stdin = open("input.txt", "r")

n, m, k, x = map(int, input().split())
distance = [0] * (n + 1)
visited = [False] * (n + 1)
road = []
for _ in range(m):
    a, b = map(int, input().split())
    road.append((a, b))


def bfs(start, road, visited, distance):
    queue = deque([start])
    visited[start] = True
    result = []
    while queue:
        v = queue.popleft()
        for a, b in road:
            if a == v:
                if not visited[b]:
                    distance[b] = distance[v] + 1
                    visited[b] = True
                    queue.append(b)
    for i in range(n + 1):
        if distance[i] == k:
            result.append(i)
    return result


result = bfs(x, road, visited, distance)
if not result:
    print(-1)
else:
    result.sort()
    for i in result:
        print(i)

```

<교재 풀이>

모든 도시에 대한 최단 거리를 distance 에 저장한다.
distance를 -1로 초기화 하고, 출발 도시까지의 거리는 0으로 초기화 한다면 visited 변수를 따로 사용하지 않아도 된다.
최단 거리가 -1인 도시는 아직 방문하지 않은 도시임을 의미한다.

```python
import sys
from collections import deque

sys.stdin = open("input.txt", "r")

n, m, k, x = map(int, input().split())
# distance 를 -1 로 초기화 한다면 visited 변수를 사용하지 않아도 된다.
distance = [-1] * (n + 1)
distance[x] = 0
graph = [[] for _ in range(n + 1)]
for _ in range(m):
    a, b = map(int, input().split())
    graph[a].append(b)

queue = deque([x])
while queue:
    now = queue.popleft()
    for next_node in graph[now]:
        # 아직 방문하지 않은 도시라면
        if distance[next_node] == -1:
            distance[next_node] = distance[now] + 1
            queue.append(next_node)

check = False
for i in range(1, n + 1):
    if distance[i] == k:
        print(i)
        check = True

if check == False:
    print(-1)

```

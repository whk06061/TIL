import sys
#sys.stdin = open("input.txt", "r")
N = int(input())
Num = list(map(int, input().split()))
res = ""
cnt = 0
tmp = 0
while N>0:
    if Num[0] > tmp and Num[N-1]> tmp:
        if Num[0] < Num[N-1]:
                res += "L"
                cnt += 1
                tmp = Num[0]
                Num.pop(0)
                N = len(Num)
        elif Num[0] > Num[N-1]:
                res += "R"
                cnt += 1
                tmp = Num[N-1]
                Num.pop()
                N = len(Num)
    elif Num[N-1] > tmp:
        res += "R"
        cnt += 1
        tmp = Num[N-1]
        Num.pop()
        N = len(Num)
    elif Num[0] > tmp:
        res += "L"
        cnt += 1
        tmp = Num[0]
        Num.pop(0)
        N = len(Num)
    else:
        break
print(cnt)
print(res)

```python
import sys
#sys.stdin = open("input.txt", "r")
N = int(input())
Num = list(map(int, input().split()))
res = ""
cnt = 0
tmp = 0
while N>0:
    if Num[0] > tmp and Num[N-1]> tmp:
        if Num[0] < Num[N-1]:
                res += "L"
                cnt += 1
                tmp = Num[0]
                Num.pop(0)
                N = len(Num)
        elif Num[0] > Num[N-1]:
                res += "R"
                cnt += 1
                tmp = Num[N-1]
                Num.pop()
                N = len(Num)
    elif Num[N-1] > tmp:
        res += "R"
        cnt += 1
        tmp = Num[N-1]
        Num.pop()
        N = len(Num)
    elif Num[0] > tmp:
        res += "L"
        cnt += 1
        tmp = Num[0]
        Num.pop(0)
        N = len(Num)
    else:
        break
print(cnt)
print(res)
```

```python
import sys
#sys.stdin = open("input.txt", "r")
n = int(input())
a = list(map(int, input().split()))
lt = 0
rt = n-1
last = 0
res = ""
tmp = []
while lt <= rt:
    if a[lt] > last:
        tmp.append((a[lt], 'L'))
    if a[rt] > last:
        tmp.append((a[rt], 'R'))
    tmp.sort()
    if len(tmp) == 0:
        break
    else:
        res = res + tmp[0][1]
        last = tmp[0][0]
        if tmp[0][1] == 'L':
            lt += 1
        else:
            rt -= 1
    tmp.clear()
print(len(res))
print(res)
```

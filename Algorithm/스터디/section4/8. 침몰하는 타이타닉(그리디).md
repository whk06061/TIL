import sys
#sys.stdin = open("input.txt", "r")
N, M = map(int, input().split())
Weight = list(map(int, input().split()))
Weight.sort(reverse=True)
sum = 0
cnt = 0
for i in range(N):
    if Weight[i] == 0:
        continue
    else:
        for j in range(i+1, N):
            if Weight[i] + Weight[j] <= M and Weight[j]!=0:
                cnt += 1
                Weight[i] = 0
                Weight[j] = 0
                break
        else:
            cnt += 1
print(cnt)

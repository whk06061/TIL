import sys
#sys.stdin = open("input.txt", "r")
s = input()
stack = []
for x in s:
    if x.isdecimal():
        stack.append(int(x))
    else:
        tmp_first = stack.pop()
        tmp_second = stack.pop()
        if x == "+":
            res = tmp_first + tmp_second
        elif x == "-":
            res = tmp_second - tmp_first
        elif x == "*":
            res = tmp_first * tmp_second
        elif x == "/":
            res = tmp_second // tmp_first
        stack.append(res)
print(res)

```python
def reverse(x):
    return int(x[::-1])

def isPrime(x):
    if x > 1:
        for i in range(2, x):
            if x % i == 0:
                return False
        else:
            return True

n = input()
input_nums = input().split()

for value in input_nums:
    x = reverse(value)
    if isPrime(x):
        print(x, end=" ")
```

```python
import sys
#sys.stdin = open("input.txt")
N = int(input())
a = []
a.append([0]*(N+2))
cnt = 0
for _ in range(N):
    tmp = [0]
    tmp += list(map(int, input().split()))
    tmp.append(0)
    a.append(tmp)
a.append([0]*(N+2))
for i in range(1,N+1):
    for j in range(1,N+1):
        cur = a[i][j]
        if cur > a[i-1][j] and cur > a[i][j-1] and cur > a[i+1][j] and cur > a[i][j+1]:
            cnt += 1
print(cnt)
```

```python
import sys
#sys.stdin = open("input.txt")
dx = [-1, 0, 1, 0]
dy = [0, 1, 0, -1]
n = int(input())
a = [list(map(int, input().split())) for _ in range(n)]
# 0으로 가장자리 둘러줌
a.insert(0, [0]*n)
a.append([0]*n)
for x in a:
    x.insert(0, 0)
    x.append(0) # x += 0 하면 어떻게 되는지 테스트
cnt = 0
for i in range(1,n+1):
    for j in range(1,n+1):
        if all(a[i][j] > a[i+dx[k]][j+dy[k]] for k in range(4)): # 이 조건이 모두 참이여야 통과
            cnt += 1
print(cnt)
```

if문의 all 함수 : all( 조건 반복문)

반복문을 돌면서 모든 조건이 만족되어야 True 반환.

<복습2>

```python
import sys
#sys.stdin = open("input.txt", "r")
n = int(input())
a = [list(map(int, input().split())) for _ in range(n)]
a.insert(0, [0]*(n+2))
a.append([0]*(n+2))
for i in range(n):
    a[i+1].insert(0,0)
    a[i+1].append(0)
dx = [-1, 1, 0, 0]
dy = [0, 0, -1, 1]
cnt = 0
for i in range(1, n+1):
    for j in range(1, n+1):
        if all(a[i][j] > a[i+dx[k]][j+dy[k]] for k in range(4)):
            cnt += 1
print(cnt)
```

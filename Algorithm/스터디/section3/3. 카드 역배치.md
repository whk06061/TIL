```python
a = []
for i in range(0, 21):
    a.append(i)
for i in range(10):
    start, end = map(int, input().split())
    for j in range(start, (start+end)//2+1):
        tmp = a[j]
        a[j] = a[end+start-j]
        a[end+start-j] = tmp
for i in range(1, 21):
    print(a[i], end=" ")
```

```python
a = []
for i in range(0, 21):
    a.append(i)
for i in range(10):
    start, end = map(int, input().split())
    tmp = a[end:start-1:-1]
    idx = 0
    for j in range(start, end+1):
        a[j] = tmp[idx]
        idx += 1
for i in range(1, 21):
    print(a[i], end=" ")
```

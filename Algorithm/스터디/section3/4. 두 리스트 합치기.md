<강의듣고 짠 코드>

```python
n1 = int(input())
a = list(map(int, input().split()))
n2 = int(input())
b = list(map(int, input().split()))
result = []
p, q = 0, 0
while True:
    if a[p] < b[q]:
        result.append(a[p])
        p += 1
        if p == n1:
            for i in b[q:]:
                result.append(i)
            break
    else:
        result.append(b[q])
        q += 1
        if q == n2:
            for i in a[p:]:
                result.append(i)
            break
for i in result:
    print(i, end=" ")
```

<강의 코드>

```python
n1 = int(input())
a = list(map(int, input().split()))
n2 = int(input())
b = list(map(int, input().split()))
result = []
p = q = 0
while p < n1 and q < n2:
    if a[p] < b[q]:
        result.append(a[p])
        p += 1
    else:
        result.append(b[q])
        q += 1
if p < n1:
	result = result + a[p:]
if q < n2:
	result = result + b[q:]

for i in result:
    print(i, end=" ")
```

<복습>

```python
import sys
#sys.stdin = open("input.txt", "rt")
n = int(input())
a = list(map(int, input().split()))
m = int(input())
b = list(map(int, input().split()))
res = []
p = q = 0
while p<n and q<m:
    if a[p] < b[q]:
        res.append(a[p])
        p += 1
    else:
        res.append(b[q])
        q += 1
if p == n:
    for i in range(q, m):
        res.append(b[i])
elif q == m:
    for i in range(p, n):
        res.append(a[i])
for x in res:
    print(x, end=" ")
```

a리스트의 p번째부터 끝까지를 res리스트에 합칠 때
for문 대신 res = res + a[p:]를 사용하면 간편하다.

import sys
#sys.stdin = open("input.txt", "r")
n = int(input())
a = list(map(int, input().split()))
m = int(input())
b = list(map(int, input().split()))
p = q = 0
res = []
while p < n and q < m:
    if a[p] < b[q]:
        res.append(a[p])
        p += 1
    else:
        res.append(b[q])
        q += 1
if p == n:
    res += b[q:]
elif q == m:
    res += a[p:]
for x in res:
    print(x, end=' ')

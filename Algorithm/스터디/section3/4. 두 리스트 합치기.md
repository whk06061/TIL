<강의듣고 짠 코드>

```python
n1 = int(input())
a = list(map(int, input().split()))
n2 = int(input())
b = list(map(int, input().split()))
result = []
p, q = 0, 0
while True:
    if a[p] < b[q]:
        result.append(a[p])
        p += 1
        if p == n1:
            for i in b[q:]:
                result.append(i)
            break
    else:
        result.append(b[q])
        q += 1
        if q == n2:
            for i in a[p:]:
                result.append(i)
            break
for i in result:
    print(i, end=" ")
```

<강의 코드>

```python
n1 = int(input())
a = list(map(int, input().split()))
n2 = int(input())
b = list(map(int, input().split()))
result = []
p = q = 0
while p < n1 and q < n2:
    if a[p] < b[q]:
        result.append(a[p])
        p += 1
    else:
        result.append(b[q])
        q += 1
if p < n1:
	result = result + a[p:]
if q < n2:
	result = result + b[q:]

for i in result:
    print(i, end=" ")
```

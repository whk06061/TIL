```python
import sys
#sys.stdin = open("input.txt", "r")
def binary(num):
    if num == 0:
        return 0
    a = num%2
    b = num//2
    return 10*binary(b)+a
n = int(input())
print(binary(n))

```

import sys
sys.stdin = open("input.txt", "r")
def DFS(x):
    if x == 0:
        return
    else:
        DFS(x//2)
        print(x%2, end='')

if __name__ == "__main__":
    n = int(input())
    DFS(n)

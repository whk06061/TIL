깊이우선탐색은 `재귀함수`로 구현하면 된다.

```python
a = list(range(1,8))
a.insert(0,0)
def preorder(x): #전위순회
    if x >= len(a):
        return
    print(a[x], end=' ')
    preorder(2*x)
    preorder(2*x+1)

def inorder(x): #중위순회
    if x>= len(a):
        return
    inorder(2*x)
    print(a[x], end=' ')
    inorder(2*x+1)

def postorder(x): #후위순회
    if x>= len(a):
        return
    postorder(2*x)
    postorder(2*x+1)
    print(a[x], end=' ')

preorder(1)
print()
inorder(1)
print()
postorder(1)
```

```python
import sys
sys.stdin = open("input.txt", "r")
def DFS(v):
    if v > 7:
        return
    else:
        #전위순회
        print(v, end=' ')
        DFS(v*2)
        DFS(v*2+1)

if __name__ == "__main__":
    DFS(1)
```
